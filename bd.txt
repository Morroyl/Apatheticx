Код для создания бд:

CREATE DATABASE WarehouseRental;
END;
GO

USE WarehouseRental;
GO

-- Таблица пользователей (без вставки данных)
CREATE TABLE Users (
    id INT PRIMARY KEY IDENTITY(1,1),
    fullName NVARCHAR(100) NOT NULL,
    email NVARCHAR(100) UNIQUE NOT NULL,
    phone NVARCHAR(20),
    passwordHash NVARCHAR(255) NOT NULL,
    role NVARCHAR(20) DEFAULT 'client', -- client, manager, director
    createdAt DATETIME DEFAULT GETDATE()
);

-- Таблица складов
CREATE TABLE Warehouses (
    id INT PRIMARY KEY IDENTITY(1,1),
    name NVARCHAR(100) NOT NULL,
    description NVARCHAR(500),
    address NVARCHAR(200) NOT NULL,
    area DECIMAL(10,2) NOT NULL,
    pricePerMonth DECIMAL(10,2) NOT NULL,
    category NVARCHAR(50),
    imageUrl NVARCHAR(500) NULL
);

-- Таблица заявок на аренду
CREATE TABLE LeaseRequests (
    id INT PRIMARY KEY IDENTITY(1,1),
    userId INT NOT NULL,
    warehouseId INT NOT NULL,
    startDate DATETIME NOT NULL,
    endDate DATETIME NOT NULL,
    status NVARCHAR(20) DEFAULT 'pending', -- pending, confirmed, cancelled, completed
    managerId INT NULL,
    createdAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (userId) REFERENCES Users(id) ON DELETE NO ACTION,
    FOREIGN KEY (warehouseId) REFERENCES Warehouses(id) ON DELETE NO ACTION,
    FOREIGN KEY (managerId) REFERENCES Users(id) ON DELETE NO ACTION
);

-- Таблица сообщений
CREATE TABLE Messages (
    id INT PRIMARY KEY IDENTITY(1,1),
    leaseRequestId INT NOT NULL,
    senderId INT NOT NULL,
    message NVARCHAR(MAX) NOT NULL,
    sentAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (leaseRequestId) REFERENCES LeaseRequests(id) ON DELETE CASCADE,
    FOREIGN KEY (senderId) REFERENCES Users(id) ON DELETE NO ACTION
);

-- Таблица отзывов
CREATE TABLE Reviews (
    id INT PRIMARY KEY IDENTITY(1,1),
    userId INT NOT NULL,
    warehouseId INT NULL,
    rating INT CHECK (rating BETWEEN 1 AND 5),
    comment NVARCHAR(1000),
    createdAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (userId) REFERENCES Users(id) ON DELETE CASCADE,
    FOREIGN KEY (warehouseId) REFERENCES Warehouses(id) ON DELETE SET NULL
);

-- Архив завершённых заявок
CREATE TABLE ArchivedLeaseRequests (
    id INT PRIMARY KEY IDENTITY(1,1),
    originalId INT NOT NULL,
    userId INT NOT NULL,
    warehouseId INT NOT NULL,
    startDate DATETIME NOT NULL,
    endDate DATETIME NOT NULL,
    status NVARCHAR(20) DEFAULT 'completed',
    managerId INT NULL,
    createdAt DATETIME DEFAULT GETDATE(),
    archivedAt DATETIME DEFAULT GETDATE()
);

-- Архив сообщений
CREATE TABLE ArchivedMessages (
    id INT PRIMARY KEY IDENTITY(1,1),
    originalId INT NOT NULL,
    leaseRequestId INT NOT NULL,
    senderId INT NOT NULL,
    message NVARCHAR(MAX) NOT NULL,
    sentAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (leaseRequestId) REFERENCES ArchivedLeaseRequests(id) ON DELETE CASCADE
);
GO
